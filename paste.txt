import React, { useContext } from 'react';
import { View, Pressable, TextInput, StyleSheet } from 'react-native';
import Ionicons from '@react-native-vector-icons/ionicons';
import { BellDot, Search } from 'lucide-react-native';
import { useNavigation, useRoute } from '@react-navigation/native';
import { SearchContext } from '../context/SearchContext';

export const HeaderNotificationUser = () => {
  const navigation = useNavigation();
  const route = useRoute();
  const isSearchScreen = route.name === 'SearchStack';
  const { query, setQuery } = useContext(SearchContext);

  return (
    <View style={{ flexDirection: 'row', alignItems: 'center' }}>
      {isSearchScreen ? (
        <TextInput
          style={styles.input}
          placeholder="Search books..."
          value={query}
          onChangeText={setQuery}
          placeholderTextColor={'#000'}
        />
      ) : (
        <Pressable
          style={{ paddingRight: 14 }}
          onPress={() => navigation.navigate('SearchStack')}
        >
          <Search color="#fff" size={30} />
        </Pressable>
      )}

      <Pressable style={{ paddingRight: 20 }}>
        <BellDot color="#fff" size={30} />
      </Pressable>
    </View>
  );
};
const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#fff',
    paddingTop: 10,
  },
  input: {
    borderWidth: 1,
    borderColor: '#ccc',
    backgroundColor: '#fff',
    marginRight: 10,
    borderRadius: 8,
    paddingHorizontal: 12,
    flex: 1,
  },
  card: {
    flex: 1 / 3,
    alignItems: 'center',
    margin: 5,
    flexDirection: 'row',
  },
  cover: {
    width: 100,
    height: 150,
    borderRadius: 8,
    backgroundColor: '#eee',
  },
  noCover: {
    justifyContent: 'center',
    alignItems: 'center',
  },
  title: {
    marginTop: 5,
    fontSize: 13,
    fontWeight: 'bold',
    textAlign: 'center',
  },
  author: {
    fontSize: 12,
    color: '#555',
    textAlign: 'center',
  },
});
